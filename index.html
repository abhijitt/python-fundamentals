Functions


def its_me(name):
    if len(name) > 2 and len(name) < 10 and name != 'Adele':
        # This is what a comment looks like
        # You can even write this instead
        # 2 < len(name) < 10 and name != 'Adele'
        print('Hi')
    elif name == 'Adele':
        """
        This is also a comment. It can span over multiple lines.
        This is equivalent of c++ /* block comment */
        """
        print('Hello')
    elif name[:3] == 'Ada':
        print('Yo')
    else:
        print('Hey')

name = input('Name:\n')
its_me(name)

Iterating on list


a = [2, 3, 5, 7, 11, 13]
for element in a:
    print('A prime number:', element)

# Iterate with index
for index in range(0, len(a)):
    print(a[index])

# How about both!
for index, value in enumerate(a):
    print(index, value)

List Comprehension


a = [1, 2, 3, 4, 5, 6, 7]

even_list = [num for num in a if num % 2 == 0]

odd_list = [num for num in a if num % 2 != 0]

matrix = [[1, 2, 3], [4, 5, 6]]
flat_matrix = [x for row in matrix for x in row]
# [1, 2, 3, 4, 5, 6]

str_int_list = ['1', '2', '3', '4', '5', '6']

int_list = [int(x) for x in str_int_list]

Common Errors

SyntaxError: invalid syntax 

for x in range (1,10)
    print(x)

Don't forget the colon!


for x in range (1,10):
    print(x)
IndentationError: unexpected indent 

for x in range (1,10):
print(x)


for x in range(0, 5):
	print(x)

Always indent with four spaces. DO NOT USE TABS


for x in range (1,10):
    print(x)
